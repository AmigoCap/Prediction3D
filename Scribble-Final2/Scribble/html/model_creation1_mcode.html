1function [MPoids, verif, err, it] = model_creation(Im1, Im2, Im3, Im4, Im5, Im6, Im7, Im8, Im9)
  2%% Donnees
  3
  4% UInt8 to binary
  5Im1 = (Im1>=128);
  6Im2 = (Im2>=128);
  7Im3 = (Im3>=128);
  8Im4 = (Im4>=128);
  9Im5 = (Im5>=128);
 10Im6 = (Im6>=128);
 11Im7 = (Im7>=128);
 12Im8 = (Im8>=128);
 13Im9 = (Im9>=128);
 14
 15% On les transforme en vecteur
 16VIm1 = double(Im1(:));
 17VIm2 = double(Im2(:));
 18VIm3 = double(Im3(:));
 19VIm4 = double(Im4(:));
 20VIm5 = double(Im5(:));
 21VIm6 = double(Im6(:));
 22VIm7 = double(Im7(:));
 23VIm8 = double(Im8(:));
 24VIm9 = double(Im9(:));
 25
 26
 27% On les met ensemble dans une matrice
 28M = [VIm1 VIm2 VIm3 VIm4 VIm5 VIm6 VIm7 VIm8 VIm9];
 29
 30%% Parametres
 31
 32% Tailles de la matrice M
 33Mlig = size(M,1);
 34Mcol = size(M,2);
 35
 36% Nombre maximal d'iterations
 37itmax = 1000;
 38
 39% Precision attendue
 40prec = 0.001;
 41
 42% Taille de la population
 43Npop = 1001;
 44
 45% Parametre pour les valeurs limites initiales dans la matrice de poids.
 46% [-Lim;Lim]
 47Lim = 10;
 48
 49% Parametre pour la fonction de seuil
 50a = 0.01;
 51
 52% Parametre pour l'intervalle de la petite mutation [x-e;x+e]
 53e = 300;
 54
 55% Proportion de mutations et croisements
 56pGM = 0.2;
 57pPM = 0.3;
 58pCr = 0.5;
 59
 60% Fonction de seuil utilisee
 61f = 1;
 62
 63%% Algorithme
 64
 65% Creation des variables
 66
 67% Population de matrices de poids
 68Pop = -Lim+2*Lim*rand(Mcol,Mlig,Npop);
 69
 70% Initialisation de la matrice resultat de l'application de la fonction 
 71% de seuil
 72Ms = zeros(Mcol, Mcol, Npop);
 73
 74% Initialisation du vecteur d'enregistrement des erreurs par iteration
 75err = zeros(Npop,1);
 76
 77% Initialisation de la premier valeur de meilleur erreur
 78% Comme on cherche la minimisation, on l'affecte la valeur infini 
 79MeilleurErr = 100;
 80
 81% Initialisation du vecteur des meilleurs erreurs de chaque iteration 
 82Merr = 100*ones(itmax,1);
 83
 84% Matrice identite pour le calcul de l'erreur
 85Ident = eye(Mcol);
 86
 87% Boucle de l'algorithme
 88j = 1;
 89while (MeilleurErr > prec && j < itmax)
 90    %% Recherche du meilleur
 91    
 92    % Application de la fonction de seuil aux matrices de poids dans la
 93    % population
 94    for i = 1:Npop
 95        Z = Pop(:,:,i)*M;
 96        Ms(:,:,i) = fseuil(Z, a, f);
 97        Ident = eye(size(M,2));
 98        err(i) = sum(sum((Ms(:,:,i)-Ident).^2));
 99    end
100    
101    % Tri descendant des erreurs trouves et aussi leurs respectives
102    % matrices de poids
103    [err, Index] = sort(err, 'descend');
104    PopTemp = zeros(size(Pop,1),size(Pop,2), size(Pop,3));
105    for i = 1:length(Index)
106        PopTemp(:,:,i) = Pop(:,:,Index(i));
107    end
108    Pop = PopTemp;
109    
110    Merr(j) = err(Npop);
111    MeilleurErr = err(Npop);
112    
113    if (MeilleurErr > prec)
114        %% Mutations
115        
116        % Petite mutation
117        
118        % On applique les mutations et croisements aux autres elements
119        % de la population
120        % Nombre d'elements de la population a subir les petites mutations
121        nPM = floor(pPM*(Npop-1));
122        
123        % Choix aleatoire des elements dans les matrices a subir les petites 
124        % mutations
125        % On applique les petites mutations au meilleur
126        elemPM = ceil(Mcol*Mlig*rand(nPM,1));
127        s = [Mcol,Mlig];
128        [I,J] = ind2sub(s, elemPM);
129
130        % Execution de la petite mutation
131        for k = 1:nPM
132            Pop(I(k),J(k),Npop-k) = Pop(I(k),J(k),Npop)-e+2*e*rand;
133        end
134        
135        % Grande mutation
136        % Nombre d'elements de la population a subir les grandes mutations
137        nGM = floor(pGM*(Npop-1));
138        
139        Pop(:,:,1:nGM) = -Lim+2*Lim*rand(Mcol,Mlig,nGM);
140        
141        % Croisements
142        % Nombre d'elements de la population a subir des croisements
143        nCr = floor(pCr*(Npop-1));
144
145        % Execution des croisements
146        for k = nGM:nGM+nCr
147            Pop(:,:,k) = (Pop(:,:,k) + Pop(:,:,Npop))/2;
148        end
149    end
150    % Incrementation
151    j = j + 1;
152end
153% Verification des iterations
154it = j-1;
155
156% Verification de l'erreur
157err = Merr(j-1);
158
159% Verification
160MPoids = Pop(:,:,Npop);
161Z = MPoids*M;
162
163verif = fseuil(Z, a, f);
164end
